141 int OPT(int i, int j)
142 {
143
144 if (i >= j - 4)
145 return 0; // No sharp turns
146 if (dp[i][j] != -1)
147 return dp[i][j]; // Memoization, if already computed return that value
148
149 int exclude = OPT(i, j - 1); // Optimal value when not including current base
150
151 int max_include = 0; // Variable to hold value of optimal pairing when including current base
152
153 // Iterating over all possibilities
154 for (int t = i; t < j - 4; t++)
155 {
156 if (canPair(s[t], s[j])) // Only check if we are pairing valid bases
157 {
158 int include = 1 + OPT(i, t - 1) + OPT(t + 1, j - 1); // Optimal value when including base upto t
159 max_include = max(max_include, include); // Comparing with maximal value
160 }
161 }
162
163 return dp[i][j] = max(exclude, max_include); // Store in memo table
164 }

204 void traceback(int i, int j)
205 {
206 if (i >= j)
207 return; // Out of bounds
208 if (dp[i][j] == dp[i][j - 1]) // Check if we need to include current or skip
209 {
210 traceback(i, j - 1); // if skip, traceback to sub-table (i,j-1)
211 return;
212 }
213
214 for (int k = i; k < j; k++) // If can't skip, look at all possibilites
215 {
216 if (canPair(s[k], s[j])) // Only check if we can pair the bases
217 {
218 int score = 0;
219 if (k - 1 >= 0)
220 score = dp[i][k - 1];
221 if (dp[i][j] == 1 + score + dp[k + 1][j - 1]) // If this is the pairing
222 {
